```
N = 8
n = 5
a = [[0 for i in range(N)] for j in range(N)]
n = int(input("input the level:"))

for i in range(1,n+1):
    j = 1
    while j<=i:
        a[i][j] = int(input())
        j += 1

print(a)

i = n -1
while i>=1:
    j = 1
    while j<=i:
        if a[i+1][j] >= a[i+1][j+1]:
            a[i][j] += a[i+1][j]
        else:
            a[i][j] += a[i+1][j+1]
        j += 1
    i -= 1

print(a[1][1])
```
